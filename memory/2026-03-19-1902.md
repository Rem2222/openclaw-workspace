def list_resources():
    resources = client.ls("viking://resources/")
    # Форматируем для таблицы
    data = []
    for r in resources:
        data.append([r['name'], r['path'], r.get('created', 'N/A')])

def search_resources(query):
    results = client.search(query, top_k=10)
    data = []
    for r in results:
        data.append([r['uri'], r.get('score', 0), r.get('snippet', '')])
    return data

def read_resource(uri, level="overview"):
